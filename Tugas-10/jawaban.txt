1. Buat Data Base 
CREATE DATABASE myshop

2. Membuat Table
    - Membuat Table users
    CREATE TABLE users(id	INTEGER AUTO INCREMENT PRIMARY KEY, name varchar(255), email varchar(255), password	varchar(255));

    - Membuat Table categories
    CREATE TABLE categories(id INTEGER AUTO INCREMENT PRIMARY KEY,name varchar(255));

    - Membuat Table items
    CREATE TABLE items ( id INTEGER AUTO INCREMENT PRIMARY KEY, name varchar(255), description varchar(255), price integer,stock integer,category_id	integer ,FOREIGN KEY (category_id) REFERENCES categories(id));
